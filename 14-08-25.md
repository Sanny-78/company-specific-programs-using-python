@problem statement-1(reversed number)
n = int(input("Enter number: "))
rev = 0
while n > 0:
    rem = n % 10
    rev = rev * 10 + rem
    n //= 10
print("Reversed number:", rev)

@problem statement-2(palindrome)
n = int(input("Enter a number: "))
temp = n
rev = 0
while temp > 0:
    digit = temp % 10
    rev = rev * 10 + digit
    temp //= 10
if n == rev:
    print("Palindrome number")
else:
    print("Not a palindrome number")
